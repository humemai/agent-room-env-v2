 285    def init_memory_systems(self, reset_semantic_decay: bool = False) -> None:
 286        """Initialize the agent's memory systems. This has nothing to do with the
 287        replay buffer.
 288
 289        Args:
 290            reset_semantic_decay: whether to reset the semantic memory system's decay
 291        """
 292        self.memory_systems = MemorySystems(
 293            episodic=EpisodicMemory(capacity=self.capacity["episodic"]),
 294            semantic=SemanticMemory(
 295                capacity=self.capacity["semantic"],
 296                decay_factor=self.semantic_decay_factor,
 297            ),
 298            short=ShortMemory(capacity=self.capacity["short"]),
 299        )
 300
 301        assert self.pretrain_semantic in [False, "exclude_walls", "include_walls"]
 302        if self.pretrain_semantic in ["exclude_walls", "include_walls"]:
 303            assert self.capacity["semantic"] > 0
 304
 305            if self.pretrain_semantic == "exclude_walls":
 306                exclude_walls = True
 307            else:
 308                exclude_walls = False
 309            room_layout = self.env.unwrapped.return_room_layout(exclude_walls)
 310
 311            _ = self.memory_systems.semantic.pretrain_semantic(
 312                semantic_knowledge=room_layout,
 313                return_remaining_space=False,
 314                freeze=False,
 315            )
 316
 317            if self.pretrain_semantic == "include_walls":
 318                assert self.memory_systems.semantic.size > 0
 319            elif "xxs" in self.env_config["room_size"]:
 320                assert self.memory_systems.semantic.size == 0
 321            else:
 322                assert self.memory_systems.semantic.size > 0
 323
 324        if reset_semantic_decay:
 325            self.num_semantic_decayed = 0

 354    def process_remaining_observations(
 355        self, remaining: list, greedy: bool
 356    ) -> tuple[list, list, list]:
 357        """Process remaining observations.
 358
 359        This is necessary since we are dealing with the observations one by one.
 360
 361        Args:
 362            remaining: observations["room"] from the environment
 363            greedy: whether to act greedily
 364
 365        Returns:
 366            states: consecutive states by taking consecutive actions.
 367            actions: actions taken (list of ints)
 368            q_values: q values (list of lists of floats)
 369
 370        """
 371        states = []
 372        actions = []
 373        q_values = []
 374        state = self.get_deepcopied_memory_state()
 375        states.append(state)
 376
 377        for obs in remaining:
 378            # select memory management action
 379            action, q_values_ = select_action(
 380                memory_types=self.memory_types["mm"],
 381                state=state,
 382                greedy=greedy,
 383                lstm=self.lstm_mm,
 384                mlp=self.mlp_mm,
 385                epsilon=self.epsilon,
 386                action_space=self.action_space["mm"],
 387            )
 388            manage_memory(
 389                self.memory_systems,
 390                self.action2str["mm"][action],
 391                split_possessive=False,
 392            )
 393            encode_observation(self.memory_systems, obs)
 394            state = self.get_deepcopied_memory_state()
 395            states.append(deepcopy(state))
 396            actions.append(action)
 397            q_values.append(q_values_)
 398
 399        assert len(states) == len(actions) + 1 == len(q_values) + 1
 400
 401        return states, actions, q_values

 403    def step_mm(
 404        self, state: dict, questions: list, greedy: bool
 405    ) -> tuple[dict, int, float, bool, list]:
 406        """Step through the environment with the memory management policy.
 407
 408        env.step() is called here. Make sure to call this function to interact with
 409        the environment.
 410
 411        Args:
 412            state: state of the memory systems
 413            questions: questions to answer
 414            greedy: whether to act greedily
 415
 416        Returns:
 417            observations: observations from the environment
 418            action: memory management action taken
 419            reward: reward received
 420            done: whether the episode is done
 421            q_values: q values (list of floats)
 422
 423        """
 424        # select memory management action
 425        action_mm, q_values = select_action(
 426            memory_types=self.memory_types["mm"],
 427            state=state,
 428            greedy=greedy,
 429            lstm=self.lstm_mm,
 430            mlp=self.mlp_mm,
 431            epsilon=self.epsilon,
 432            action_space=self.action_space["mm"],
 433        )
 434        manage_memory(
 435            self.memory_systems,
 436            self.action2str["mm"][action_mm],
 437            split_possessive=False,
 438        )
 439
 440        action_explore = explore(self.memory_systems, self.explore_policy_heuristic)
 441        answers = [
 442            answer_question(
 443                self.memory_systems, self.qa_function, q, split_possessive=False
 444            )
 445            for q in questions
 446        ]
 447        (
 448            observations,
 449            reward,
 450            done,
 451            truncated,
 452            info,
 453        ) = self.env.step((answers, action_explore))
 454        if (
 455            hasattr(self.memory_systems, "semantic")
 456            and self.memory_systems.semantic.capacity > 0
 457        ):
 458            self.memory_systems.semantic.decay()
 459            self.num_semantic_decayed += 1
 460        done = done or truncated
 461
 462        return observations, action_mm, reward, done, q_values

 464    def fill_replay_buffer_mm(self) -> None:
 465        """Make the replay buffer full in the beginning with the uniformly-sampled
 466        actions. The filling continues until it reaches the warm start size.
 467
 468        """
 469        self.replay_buffer = ReplayBuffer(
 470            observation_type="dict",
 471            size=self.replay_buffer_size,
 472            batch_size=self.batch_size,
 473        )
 474
 475        new_episode_starts = True
 476        while len(self.replay_buffer) < self.warm_start:
 477
 478            if new_episode_starts:
 479                self.init_memory_systems(reset_semantic_decay=True)
 480                observations, info = self.env.reset()
 481                done = False
 482                remaining = self.process_first_observation(observations["room"])
 483                new_episode_starts = False
 484
 485            else:
 486                state = self.get_deepcopied_memory_state()
 487                observations, action, reward, done, q_values = self.step_mm(
 488                    state, observations["questions"], greedy=False
 489                )
 490                remaining = self.process_first_observation(observations["room"])
 491                next_state = self.get_deepcopied_memory_state()
 492                self.replay_buffer.store(*[state, action, reward, next_state, done])
 493
 494            if done:
 495                new_episode_starts = True
 496
 497            else:
 498                states_, actions_, q_values_ = self.process_remaining_observations(
 499                    remaining, greedy=False
 500                )
 501                reward = 0
 502                done = False
 503                for state, next_state, action in zip(
 504                    states_[:-1], states_[1:], actions_
 505                ):
 506                    self.replay_buffer.store(*[state, action, reward, next_state, done])

 508    def train_mm(self) -> None:
 509        """Train the memory management policy."""
 510        os.makedirs(os.path.join(self.default_root_dir, "mm"), exist_ok=True)
 511        self.optimizer = optim.Adam(
 512            list(self.lstm_mm.parameters()) + list(self.mlp_mm.parameters())
 513        )
 514
 515        self.fill_replay_buffer_mm()
 516        self.epsilon = self.max_epsilon
 517
 518        self.lstm_mm.train()
 519        self.mlp_mm.train()
 520
 521        new_episode_starts = True
 522        score = 0
 523
 524        self.iteration_idx = 0
 525
 526        while True:
 527            if new_episode_starts:
 528                self.init_memory_systems(reset_semantic_decay=True)
 529                observations, info = self.env.reset()
 530                done = False
 531                remaining = self.process_first_observation(observations["room"])
 532                new_episode_starts = False
 533
 534            else:
 535                state = self.get_deepcopied_memory_state()
 536                observations, action, reward, done, q_values = self.step_mm(
 537                    state, observations["questions"], greedy=False
 538                )
 539                remaining = self.process_first_observation(observations["room"])
 540                next_state = self.get_deepcopied_memory_state()
 541                self.replay_buffer.store(*[state, action, reward, next_state, done])
 542                self.q_values["mm"]["train"].append(q_values)
 543                score += reward
 544                self.iteration_idx += 1
 545
 546            if done:
 547                new_episode_starts = True
 548                if (
 549                    hasattr(self.memory_systems, "semantic")
 550                    and self.memory_systems.semantic.capacity > 0
 551                ):
 552                    assert (
 553                        self.num_semantic_decayed
 554                        == self.env_config["terminates_at"] + 1
 555                    )
 556                self.scores["mm"]["train"].append(score)
 557                score = 0
 558
 559                if (
 560                    self.iteration_idx >= self.validation_starts_at
 561                ) and self.iteration_idx % (
 562                    self.validation_interval * (self.env_config["terminates_at"] + 1)
 563                ) == 0:
 564                    with torch.no_grad():
 565                        self.validate(self.lstm_mm, self.mlp_mm, "mm")
 566
 567            else:
 568                states_, actions_, q_values_ = self.process_remaining_observations(
 569                    remaining, greedy=False
 570                )
 571                reward = 0
 572                for state, next_state, action in zip(
 573                    states_[:-1], states_[1:], actions_
 574                ):
 575                    self.replay_buffer.store(*[state, action, reward, next_state, done])
 576                self.q_values["mm"]["train"].extend(q_values_)
 577
 578            if not new_episode_starts:
 579                loss = update_model(
 580                    memory_types=self.memory_types["mm"],
 581                    replay_buffer=self.replay_buffer,
 582                    optimizer=self.optimizer,
 583                    device=self.device,
 584                    lstm=self.lstm_mm,
 585                    lstm_target=self.lstm_mm_target,
 586                    mlp=self.mlp_mm,
 587                    mlp_target=self.mlp_mm_target,
 588                    ddqn=self.ddqn,
 589                    gamma=self.gamma["mm"],
 590                )
 591
 592                self.training_loss["mm"].append(loss)
 593
 594                # linearly decay epsilon
 595                self.epsilon = update_epsilon(
 596                    self.epsilon,
 597                    self.max_epsilon,

 286    def init_memory_systems(self, reset_semantic_decay: bool = False) -> None:
 287        """Initialize the agent's memory systems. This has nothing to do with the
 288        replay buffer.
 289
 290        Args:
 291            reset_semantic_decay: whether to reset the semantic memory system's decay
 292        """
 293        self.memory_systems = MemorySystems(
 294            episodic=EpisodicMemory(capacity=self.capacity["episodic"]),
 295            semantic=SemanticMemory(
 296                capacity=self.capacity["semantic"],
 297                decay_factor=self.semantic_decay_factor,
 298            ),
 299            short=ShortMemory(capacity=self.capacity["short"]),
 300        )
 301
 302        assert self.pretrain_semantic in [False, "exclude_walls", "include_walls"]
 303        if self.pretrain_semantic in ["exclude_walls", "include_walls"]:
 304            assert self.capacity["semantic"] > 0
 305
 306            if self.pretrain_semantic == "exclude_walls":
 307                exclude_walls = True
 308            else:
 309                exclude_walls = False
 310            room_layout = self.env.unwrapped.return_room_layout(exclude_walls)
 311
 312            _ = self.memory_systems.semantic.pretrain_semantic(
 313                semantic_knowledge=room_layout,
 314                return_remaining_space=False,
 315                freeze=False,
 316            )
 317
 318            if self.pretrain_semantic == "include_walls":
 319                assert self.memory_systems.semantic.size > 0
 320            elif "xxs" in self.env_config["room_size"]:
 321                assert self.memory_systems.semantic.size == 0
 322            else:
 323                assert self.memory_systems.semantic.size > 0
 324
 325        if reset_semantic_decay:
 326            self.num_semantic_decayed = 0

 339    def process_first_observation(self, observations_room: list) -> list:
 340        """Encode the first observation into the short-term memory system.
 341
 342        This is necessary since we are dealing with the observations one by one.
 343
 344        Args:
 345            observations_room: observations["room"] from the environment
 346
 347        Returns:
 348            observations_room[1:]: remaining observations
 349
 350        """
 351        encode_observation(self.memory_systems, observations_room[0])
 352
 353        return observations_room[1:]

 355    def process_remaining_observations(
 356        self, remaining: list, greedy: bool
 357    ) -> tuple[list, list, list]:
 358        """Process remaining observations.
 359
 360        This is necessary since we are dealing with the observations one by one.
 361
 362        Args:
 363            remaining: observations["room"] from the environment
 364            greedy: whether to act greedily
 365
 366        Returns:
 367            states: consecutive states by taking consecutive actions.
 368            actions: actions taken (list of ints)
 369            q_values: q values (list of lists of floats)
 370
 371        """
 372        states = []
 373        actions = []
 374        q_values = []
 375        state = self.get_deepcopied_memory_state()
 376        states.append(state)
 377
 378        for obs in remaining:
 379            # select memory management action
 380            action, q_values_ = select_action(
 381                memory_types=self.memory_types["mm"],
 382                state=state,
 383                greedy=greedy,
 384                lstm=self.lstm_mm,
 385                mlp=self.mlp_mm,
 386                epsilon=self.epsilon,
 387                action_space=self.action_space["mm"],
 388            )
 389            manage_memory(
 390                self.memory_systems,
 391                self.action2str["mm"][action],
 392                split_possessive=False,
 393            )
 394            encode_observation(self.memory_systems, obs)
 395            state = self.get_deepcopied_memory_state()
 396            states.append(deepcopy(state))
 397            actions.append(action)
 398            q_values.append(q_values_)
 399
 400        assert len(states) == len(actions) + 1 == len(q_values) + 1
 401
 402        return states, actions, q_values

 404    def step_mm(
 405        self, state: dict, questions: list, greedy: bool
 406    ) -> tuple[dict, int, float, bool, list]:
 407        """Step through the environment with the memory management policy.
 408
 409        env.step() is called here. Make sure to call this function to interact with
 410        the environment.
 411
 412        Args:
 413            state: state of the memory systems
 414            questions: questions to answer
 415            greedy: whether to act greedily
 416
 417        Returns:
 418            observations: observations from the environment
 419            action: memory management action taken
 420            reward: reward received
 421            done: whether the episode is done
 422            q_values: q values (list of floats)
 423
 424        """
 425        # select memory management action
 426        action_mm, q_values = select_action(
 427            memory_types=self.memory_types["mm"],
 428            state=state,
 429            greedy=greedy,
 430            lstm=self.lstm_mm,
 431            mlp=self.mlp_mm,
 432            epsilon=self.epsilon,
 433            action_space=self.action_space["mm"],
 434        )
 435        manage_memory(
 436            self.memory_systems,
 437            self.action2str["mm"][action_mm],
 438            split_possessive=False,
 439        )
 440
 441        action_explore = explore(self.memory_systems, self.explore_policy_heuristic)
 442        answers = [
 443            answer_question(
 444                self.memory_systems, self.qa_function, q, split_possessive=False
 445            )
 446            for q in questions
 447        ]
 448        (
 449            observations,
 450            reward,
 451            done,
 452            truncated,
 453            info,
 454        ) = self.env.step((answers, action_explore))
 455        if (
 456            hasattr(self.memory_systems, "semantic")
 457            and self.memory_systems.semantic.capacity > 0
 458        ):
 459            self.memory_systems.semantic.decay()
 460            self.num_semantic_decayed += 1
 461        done = done or truncated
 462
 463        return observations, action_mm, reward, done, q_values

 465    def fill_replay_buffer_mm(self) -> None:
 466        """Make the replay buffer full in the beginning with the uniformly-sampled
 467        actions. The filling continues until it reaches the warm start size.
 468
 469        """
 470        self.replay_buffer = ReplayBuffer(
 471            observation_type="dict",
 472            size=self.replay_buffer_size,
 473            batch_size=self.batch_size,
 474        )
 475
 476        new_episode_starts = True
 477        while len(self.replay_buffer) < self.warm_start:
 478
 479            if new_episode_starts:
 480                self.init_memory_systems(reset_semantic_decay=True)
 481                observations, info = self.env.reset()
 482                done = False
 483                remaining = self.process_first_observation(observations["room"])
 484                new_episode_starts = False
 485
 486            else:
 487                state = self.get_deepcopied_memory_state()
 488                observations, action, reward, done, q_values = self.step_mm(
 489                    state, observations["questions"], greedy=False
 490                )
 491                remaining = self.process_first_observation(observations["room"])
 492                next_state = self.get_deepcopied_memory_state()
 493                self.replay_buffer.store(*[state, action, reward, next_state, done])
 494
 495            if done:
 496                new_episode_starts = True
 497
 498            else:
 499                states_, actions_, q_values_ = self.process_remaining_observations(
 500                    remaining, greedy=False
 501                )
 502                reward = 0
 503                done = False
 504                for state, next_state, action in zip(
 505                    states_[:-1], states_[1:], actions_
 506                ):
 507                    self.replay_buffer.store(*[state, action, reward, next_state, done])
 508
 509    def train_mm(self) -> None:
 510        """Train the memory management policy."""
 511        os.makedirs(os.path.join(self.default_root_dir, "mm"), exist_ok=True)
 512        self.optimizer = optim.Adam(
 513            list(self.lstm_mm.parameters()) + list(self.mlp_mm.parameters())
 514        )
 515
 516        self.fill_replay_buffer_mm()
 517        self.epsilon = self.max_epsilon
 518
 519        self.lstm_mm.train()
 520        self.mlp_mm.train()
 521
 522        new_episode_starts = True
 523        score = 0
 524
 525        self.iteration_idx = 0
 526
 527        while True:
 528            if new_episode_starts:
 529                self.init_memory_systems(reset_semantic_decay=True)
 530                observations, info = self.env.reset()
 531                done = False
 532                remaining = self.process_first_observation(observations["room"])
 533                new_episode_starts = False
 534
 535            else:
 536                state = self.get_deepcopied_memory_state()
 537                observations, action, reward, done, q_values = self.step_mm(
 538                    state, observations["questions"], greedy=False
 539                )
 540                remaining = self.process_first_observation(observations["room"])
 541                next_state = self.get_deepcopied_memory_state()
 542                self.replay_buffer.store(*[state, action, reward, next_state, done])
 543                self.q_values["mm"]["train"].append(q_values)
 544                score += reward
 545                self.iteration_idx += 1
 546
 547            if done:
 548                new_episode_starts = True
 549                if (
 550                    hasattr(self.memory_systems, "semantic")
 551                    and self.memory_systems.semantic.capacity > 0
 552                ):
 553                    assert (
 554                        self.num_semantic_decayed
 555                        == self.env_config["terminates_at"] + 1
 556                    )
 557                self.scores["mm"]["train"].append(score)
 558                score = 0
 559
 560                if (
 561                    self.iteration_idx >= self.validation_starts_at
 562                ) and self.iteration_idx % (
 563                    self.validation_interval * (self.env_config["terminates_at"] + 1)
 564                ) == 0:
 565                    with torch.no_grad():
 566                        self.validate(self.lstm_mm, self.mlp_mm, "mm")
 567
 568            else:
 569                states_, actions_, q_values_ = self.process_remaining_observations(
 570                    remaining, greedy=False
 571                )
 572                reward = 0
 573                for state, next_state, action in zip(
 574                    states_[:-1], states_[1:], actions_
 575                ):
 576                    self.replay_buffer.store(*[state, action, reward, next_state, done])
 577                self.q_values["mm"]["train"].extend(q_values_)
 578
 579            if not new_episode_starts:
 580                loss = update_model(
 581                    memory_types=self.memory_types["mm"],
 582                    replay_buffer=self.replay_buffer,
 583                    optimizer=self.optimizer,
 584                    device=self.device,
 585                    lstm=self.lstm_mm,
 586                    lstm_target=self.lstm_mm_target,
 587                    mlp=self.mlp_mm,
 588                    mlp_target=self.mlp_mm_target,
 589                    ddqn=self.ddqn,
 590                    gamma=self.gamma["mm"],
 591                )
 592
 593                self.training_loss["mm"].append(loss)
 594
 595                # linearly decay epsilon
 596                self.epsilon = update_epsilon(
 597                    self.epsilon,
 598                    self.max_epsilon,
 599                    self.min_epsilon,
 600                    self.epsilon_decay_until,
 601                )
 602                self.epsilons["mm"].append(self.epsilon)
 603
 604                # if hard update is needed
 605                if self.iteration_idx % self.target_update_interval == 0:
 606                    target_hard_update(self.lstm_mm, self.lstm_mm_target)
 607                    target_hard_update(self.mlp_mm, self.mlp_mm_target)
 608
 609                # plotting & show training results
 610                if (
 611                    self.iteration_idx == self.num_iterations
 612                    or self.iteration_idx % self.plotting_interval == 0
 613                ):
 614                    self.plot_results("mm", "all", save_fig=True)
 615
 616                if self.iteration_idx == self.num_iterations:
 617                    break
 618
 619        with torch.no_grad():
 620            self.test(self.lstm_mm, self.mlp_mm, "mm")
 621
 622        self.env.close()

 624    def validate_test_middle_mm(
 625        self, val_or_test: str
 626    ) -> tuple[list, list, list, list]:
 627        """A function shared by validation and test in the middle.
 628
 629        Args:
 630            val_or_test: "val" or "test"
 631
 632        Returns:
 633            scores_local: a list of total episode rewards
 634            states_local: memory states
 635            q_values_local: q values
 636            actions_local: greey actions taken
 637
 638        """
 639        scores_local = []
 640        states_local = []
 641        q_values_local = []
 642        actions_local = []
 643
 644        for idx in range(self.num_samples_for_results[val_or_test]):
 645            new_episode_starts = True
 646            score = 0
 647            while True:
 648                if new_episode_starts:
 649                    self.init_memory_systems(reset_semantic_decay=True)
 650                    observations, info = self.env.reset()
 651                    done = False
 652                    remaining = self.process_first_observation(observations["room"])
 653                    new_episode_starts = False
 654
 655                else:
 656                    state = self.get_deepcopied_memory_state()
 657                    observations, action, reward, done, q_values = self.step_mm(
 658                        state, observations["questions"], greedy=True
 659                    )
 660                    remaining = self.process_first_observation(observations["room"])
 661                    score += reward
 662
 663                    if idx == self.num_samples_for_results[val_or_test] - 1:
 664                        states_local.append(state)
 665                        q_values_local.append(q_values)
 666                        self.q_values["mm"][val_or_test].append(q_values)
 667                        actions_local.append(action)
 668
 669                if done:
 670                    if (
 671                        hasattr(self.memory_systems, "semantic")
 672                        and self.memory_systems.semantic.capacity > 0
 673                    ):
 674                        assert (
 675                            self.num_semantic_decayed
 676                            == self.env_config["terminates_at"] + 1
 677                        )
 678                    break
 679
 680                else:
 681                    states_, actions_, q_values_ = self.process_remaining_observations(
 682                        remaining, greedy=True
 683                    )
 684
 685                    if idx == self.num_samples_for_results[val_or_test] - 1:
 686                        states_local.extend(states_[:-1])
 687                        q_values_local.extend(q_values_)
 688                        self.q_values["mm"][val_or_test].extend(q_values_)
 689                        actions_local.extend(actions_)
 690
 691            scores_local.append(score)
 692
 693        return scores_local, states_local, q_values_local, actions_local

 697        lstm: torch.nn.Module,
 698        mlp: torch.nn.Module,
 699        policy_type: Literal["mm", "explore"],

 708        """

 710        mlp.eval()

 712        if policy_type == "mm":

 714        elif policy_type == "explore":
 715            middle_function = self.validate_test_middle_explore
 716        else:
 717            raise ValueError("policy_type should be either 'mm' or 'explore'")
 718

 740            "val",
 741            num_episodes,

 745        mlp.train()

 23class LSTM(torch.nn.Module):
 24    """LSTMMLP for the DQN baseline."""
 25
 26    def __init__(
 27        self,
 28        entities: list,
 29        relations: list,
 30        num_layers: int = 2,
 31        embedding_dim: int = 64,
 32        hidden_size: int = 64,
 33        bidirectional: bool = False,
 34        device: str = "cpu",
 35        use_one_hot: bool = False,
 36    ) -> None:
 37        """Initialize the LSTM.
 38
 39        Args:
 40            entities: List of entities.
 41            relations: List of relations.
 42            num_layers: Number of layers in the LSTM.
 43            embedding_dim: Dimension of the embeddings.
 44            hidden_size: Hidden size of the LSTM.
 45            bidirectional: Whether the LSTM is bidirectional.
 46            device: Device to use.
 47            use_one_hot: Whether to use one-hot encoding.
 48
 49        """
 50        super().__init__()
 51        self.entities = entities
 52        self.relations = relations
 53        self.hidden_size = hidden_size
 54        self.num_layers = num_layers
 55        self.embedding_dim = embedding_dim
 56        self.bidirectional = bidirectional
 57        self.device = device
 58        self.use_one_hot = use_one_hot
 59
 60        self.create_embeddings()
 61        self.lstm = torch.nn.LSTM(
 62            self.embedding_dim if not use_one_hot else len(self.word2idx),
 63            self.hidden_size,
 64            self.num_layers,
 65            batch_first=True,
 66            bidirectional=bidirectional,
 67            device=self.device,
 68        )
 69
 70    def create_embeddings(self) -> None:
 71        """Create learnable embeddings."""
 72        if isinstance(self.entities, dict):
 73            self.word2idx = (
 74                ["<PAD>"]
 75                + [name for names in self.entities.values() for name in names]
 76                + self.relations
 77            )
 78        elif isinstance(self.entities, list):
 79            self.word2idx = ["<PAD>"] + self.entities + self.relations
 80        else:
 81            raise ValueError(
 82                "entities should be either a list or a dictionary, but "
 83                f"{type(self.entities)} was given!"
 84            )
 85        self.word2idx = {word: idx for idx, word in enumerate(self.word2idx)}
 86
 87        if not self.use_one_hot:
 88            self.embeddings = nn.Embedding(
 89                len(self.word2idx),
 90                self.embedding_dim,
 91                device=self.device,
 92                padding_idx=0,
 93            )
 94
 95    def get_one_hot(self, idx: int, length: int) -> torch.Tensor:
 96        """Get one-hot encoding for a given index.
 97
 98        Args:
 99            idx: index of the word
100            length: length of the one-hot vector
101
102        Returns:
103            one-hot encoding of the given index.
104        """
105        one_hot = torch.zeros(length, device=self.device)
106        one_hot[idx] = 1.0
107        return one_hot
108
109    def make_embedding(self, obs: list[str]) -> torch.Tensor:
110        """Create one embedding vector with summation.
111
112        Args:
113            obs: Observation.
114
115        Returns:
116            final_embedding: Final embedding.
117
118        """
119        if obs == ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]:
120            if self.use_one_hot:
121                return self.get_one_hot(self.word2idx["<PAD>"], len(self.word2idx))
122            return self.embeddings(
123                torch.tensor(self.word2idx["<PAD>"], device=self.device)
124            )
125
126        head_embedding = (
127            self.get_one_hot(self.word2idx[obs[0]], len(self.word2idx))
128            if self.use_one_hot
129            else self.embeddings(
130                torch.tensor(self.word2idx[obs[0]], device=self.device)
131            )
132        )
133        relation_embedding = (
134            self.get_one_hot(self.word2idx[obs[1]], len(self.word2idx))
135            if self.use_one_hot
136            else self.embeddings(
137                torch.tensor(self.word2idx[obs[1]], device=self.device)
138            )
139        )
140        tail_embedding = (
141            self.get_one_hot(self.word2idx[obs[2]], len(self.word2idx))
142            if self.use_one_hot
143            else self.embeddings(
144                torch.tensor(self.word2idx[obs[2]], device=self.device)
145            )
146        )
147        final_embedding = head_embedding + relation_embedding + tail_embedding
148
149        return final_embedding
150
151    def forward(
152        self, x: np.ndarray, memory_types: None = None
153    ) -> tuple[torch.Tensor, None]:
154        """Forward pass.
155
156        Args:
157            x: Input.
158
159        Returns:
160            lstm_out: LSTM last hidden state.
161            None: This is for compatibility with the DQN agent.
162
163        """
164        obs_pad = ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]
165        batch = [sample["data"] for sample in x]
166        max_len = 0
167        for sample in batch:
168            if len(sample) > max_len:
169                max_len = len(sample)
170
171        for observations in batch:
172            while len(observations) < max_len:
173                observations.append(obs_pad)
174
175        batch_embeddings = []
176        for observations in batch:
177            embeddings = []
178            for obs in observations:
179                obs_emb = self.make_embedding(obs)
180                embeddings.append(obs_emb)
181            embeddings = torch.stack(embeddings)
182            batch_embeddings.append(embeddings)
183
184        batch_embeddings = torch.stack(batch_embeddings)
185
186        lstm_out = self.lstm(batch_embeddings)[0]
187        lstm_out = lstm_out[:, -1, :]
188
189        return lstm_out, None

192class MLP(torch.nn.Module):
193    """Multi-layer perceptron with ReLU activation functions."""
194
195    def __init__(
196        self,
197        n_actions: int,
198        hidden_size: int,
199        device: str,
200        num_hidden_layers: int = 1,
201        dueling_dqn: bool = True,
202    ) -> None:
203        """Initialize the MLP.
204
205        Args:
206            n_actions: Number of actions.
207            hidden_size: Hidden size of the linear layer.
208            device: "cpu" or "cuda".
209            num_hidden_layers: int, number of hidden layers in the MLP.
210            dueling_dqn: Whether to use dueling DQN.
211
212        """
213        super(MLP, self).__init__()
214        self.device = device
215        self.hidden_size = hidden_size
216        self.num_hidden_layers = num_hidden_layers
217        self.n_actions = n_actions
218        self.dueling_dqn = dueling_dqn
219
220        # Define the layers for the advantage stream
221        advantage_layers = []
222        for _ in range(self.num_hidden_layers):
223            advantage_layers.append(
224                torch.nn.Linear(self.hidden_size, self.hidden_size, device=self.device)
225            )
226            advantage_layers.append(torch.nn.ReLU())
227        advantage_layers.append(
228            torch.nn.Linear(self.hidden_size, self.n_actions, device=self.device)
229        )
230        self.advantage_layer = torch.nn.Sequential(*advantage_layers)
231
232        if self.dueling_dqn:
233            # Define the layers for the value stream
234            value_layers = []
235            for _ in range(self.num_hidden_layers):
236                value_layers.append(
237                    torch.nn.Linear(
238                        self.hidden_size, self.hidden_size, device=self.device
239                    )
240                )
241                value_layers.append(torch.nn.ReLU())
242            value_layers.append(
243                torch.nn.Linear(self.hidden_size, 1, device=self.device)
244            )
245            self.value_layer = torch.nn.Sequential(*value_layers)
246
247    def forward(self, x: torch.Tensor) -> torch.Tensor:
248        """Forward pass of the neural network.
249
250        Args:
251            x: Input tensor. The shape is (batch_size, lstm_hidden_size).
252        Returns:
253            torch.Tensor: Output tensor. The shape is (batch_size, n_actions).
254
255        """
256
257        if self.dueling_dqn:
258            value = self.value_layer(x)
259            advantage = self.advantage_layer(x)
260            q = value + advantage - advantage.mean(dim=-1, keepdim=True)
261        else:
262            q = self.advantage_layer(x)
263
264        return q

267class History:
268    def __init__(
269        self,
270        block_size: int = 4,
271        action2str: dict = {0: "north", 1: "east", 2: "south", 3: "west", 4: "stay"},
272    ) -> None:
273        """Initialize the history.
274
275        Args:
276            block_size: Block size.
277            action2str: Action number to action string.
278
279        """
280        self.block_size = block_size
281        self.blocks = [[]] * self.block_size
282
283        self.action2str = action2str
284        self.actions_str = list(self.action2str.values())
285        self.actions_int = list(self.action2str.keys())
286
287    def to_list(self) -> list:
288        """Convert the history to a list."""
289        return [element for block in self.blocks for element in block]
290
291    def add_block(self, block: list) -> None:
292        """Add a block to the history. The oldest block is removed."""
293        self.blocks = self.blocks[1:] + [block]
294
295    def __repr__(self) -> str:
296        return str(self.blocks)
297
298    def answer_question(self, question: list) -> str:
299        """Answer a question, by going through the history in backwards. This means that
300        the most recent and relevant observation is used to answer the question.
301        """
302        assert len(question) == 4 and question[2] == "?"
303        for obs in self.to_list()[::-1]:
304            if obs[0] == question[0] and obs[1] == question[1]:
305                return obs[2]
306
307    def answer_questions(self, questions: list[list[str]]) -> list[str]:
308        """Answer a list of questions.
309
310        Args:
311            questions: List of questions.
312
313        Returns:
314            answers: List of answers.
315        """
316        return [self.answer_question(question) for question in questions]
317
318    def find_agent_current_location(self) -> str | None:
319        """Find the current location of the agent, by going through the history in
320        backwards."""
321        for obs in self.to_list()[::-1]:
322            if obs[0] == "agent" and obs[1] == "atlocation":
323                return obs[2]
324
325        return None
326
327    def explore(self, explore_policy: str) -> str:
328        """Explore the environment. The agent can either explore randomly or avoid
329        walls. This is not reinforcement learning, but a simple heuristic.
330
331        Args:
332            explore_policy: Explore policy.
333
334        Returns:
335            action_explore: Action to explore.
336
337        """
338        if explore_policy not in ["random", "avoid_walls"]:
339            raise ValueError(f"Unknown explore policy: {explore_policy}")
340
341        if explore_policy == "random":
342            action_explore = random.choice(self.actions_str)
343        elif explore_policy == "avoid_walls":
344            current_agent_location = self.find_agent_current_location()
345            if current_agent_location is None:
346                action_explore = random.choice(self.actions_str)
347            else:
348                relevant_observations = []
349                relevant_observations += [
350                    obs
351                    for obs in self.to_list()[::-1]
352                    if obs[0] == current_agent_location
353                    and obs[1] in ["north", "east", "south", "west"]
354                ]
355                # we know the agent's current location but there is no memory about map
356                if len(relevant_observations) == 0:
357                    action_explore = random.choice(self.actions_str)
358                else:
359                    # we know the agent's current location and there is at least one
360                    # memory about the map and we want to avoid the walls
361
362                    to_take = []
363                    to_avoid = []
364
365                    for obs in relevant_observations:
366                        if isinstance(obs, dict):
367                            continue
368                        if obs[2].split("_")[0] == "room":
369                            to_take.append(obs[1])
370                        elif obs[2] == "wall":
371                            if obs[1] not in to_avoid:
372                                to_avoid.append(obs[1])
373
374                    if len(to_take) > 0:
375                        action_explore = random.choice(to_take)
376                    else:
377                        options = deepcopy(self.actions_str)
378                        for e in to_avoid:
379                            options.remove(e)
380
381                        action_explore = random.choice(options)
382
383        return action_explore

543    def _run_explore_baselines(self) -> None:

553        results = {}
554        for explore_policy in ["random", "avoid_walls"]:
555            results[str((explore_policy))] = []
556            scores = []
557            for test_seed in [0, 1, 2, 3, 4]:
558                score = 0

573                        observations,

620    def step(

622    ) -> tuple[dict, int, float, bool, list]:
623        """Step through the environment.
624

626        function to interact with the environment.
627
628        Args:
629            state: state of the history
630            questions: questions to answer
631            greedy: whether to act greedily
632
633        Returns:
634            observations: observations from the environment
635            action_explore: exploration action taken
636            reward: reward received
637            done: whether the episode is done
638            q_values: q values (list of floats)

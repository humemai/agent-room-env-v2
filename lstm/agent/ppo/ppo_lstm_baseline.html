 37        self,
 38        entities: list,
 39        relations: list,
 40        n_actions: int,
 41        max_step_reward: int,

 43        num_layers: int = 2,
 44        embedding_dim: int = 64,
 45        make_categorical_embeddings: bool = False,
 46        batch_first: bool = True,

 56

 58            entities: List of entities.
 59            relations: List of relations.
 60            n_actions: Number of actions.

 63            num_layers: Number of layers in the LSTM.

 76        super().__init__()
 77        self.entities = entities
 78        self.relations = relations
 79        self.n_actions = n_actions
 80        self.max_step_reward = max_step_reward
 81        self.hidden_size = hidden_size
 82        self.num_layers = num_layers

147    def create_embeddings(self) -> None:
148        """Create learnable embeddings."""
149        if isinstance(self.entities, dict):
150            self.word2idx = (
151                ["<PAD>"]
152                + [name for names in self.entities.values() for name in names]
153                + self.relations
154            )
155        elif isinstance(self.entities, list):
156            self.word2idx = ["<PAD>"] + self.entities + self.relations
157        else:
158            raise ValueError(
159                "entities should be either a list or a dictionary, but "
160                f"{type(self.entities)} was given!"
161            )
162        self.word2idx = {word: idx for idx, word in enumerate(self.word2idx)}
163
164        self.embeddings = torch.nn.Embedding(
165            len(self.word2idx),
166            self.embedding_dim,
167            device=self.device,
168            padding_idx=0,
169        )
170
171        if self.make_categorical_embeddings:
172            # Assuming self.entities is a dictionary where keys are categories and
173            # values are lists of entity names
174
175            # Create a dictionary to keep track of starting index for each category
176            category_start_indices = {}
177            current_index = 1  # Start from 1 to skip the <PAD> token
178            for category, names in self.entities.items():
179                category_start_indices[category] = current_index
180                current_index += len(names)
181
182            # Re-initialize embeddings by category
183            for category, start_idx in category_start_indices.items():
184                end_idx = start_idx + len(self.entities[category])
185                init_vector = torch.randn(self.embedding_dim, device=self.device)
186                self.embeddings.weight.data[start_idx:end_idx] = init_vector.repeat(
187                    end_idx - start_idx, 1
188                )
189            # Note: Relations are not re-initialized by category, assuming they are
190            # separate from entities
191
192        if self.fuse_information == "concat":
193            self.input_size = self.embedding_dim * 3
194        elif self.fuse_information == "sum":
195            self.input_size = self.embedding_dim
196        else:
197            raise ValueError(
198                f"fuse_information should be one of 'concat' or 'sum', but "
199                f"{self.fuse_information} was given!"
200            )
201
202    def make_embedding(self, obs: list[str]) -> torch.Tensor:
203        """Create one embedding vector with summation or concatenation.
204
205        Args:
206            obs: Observation.
207
208        Returns:
209            final_embedding: Final embedding.
210
211        """
212        if obs == ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]:
213            if self.fuse_information == "sum":
214                return self.embeddings(
215                    torch.tensor(self.word2idx["<PAD>"], device=self.device)
216                )
217            else:
218                final_embedding = torch.concat(
219                    [
220                        self.embeddings(
221                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
222                        ),
223                        self.embeddings(
224                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
225                        ),
226                        self.embeddings(
227                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
228                        ),
229                    ]
230                )
231                return final_embedding
232
233        head_embedding = self.embeddings(
234            torch.tensor(self.word2idx[obs[0]], device=self.device)
235        )
236        relation_embedding = self.embeddings(
237            torch.tensor(self.word2idx[obs[1]], device=self.device)
238        )
239        tail_embedding = self.embeddings(
240            torch.tensor(self.word2idx[obs[2]], device=self.device)
241        )
242        if self.fuse_information == "concat":
243            final_embedding = torch.concat(
244                [head_embedding, relation_embedding, tail_embedding]
245            )
246        elif self.fuse_information == "sum":
247            final_embedding = head_embedding + relation_embedding + tail_embedding
248        else:
249            raise ValueError(
250                f"fuse_information should be one of 'concat' or 'sum', but "
251                f"{self.fuse_information} was given!"
252            )
253
254        if self.include_positional_encoding:
255            final_embedding += self.positional_encoding[obs[3]]
256
257        return final_embedding

259    def forward(self, x: np.ndarray) -> torch.Tensor:
260        """Forward pass.
261
262        Args:
263            x: Input.
264
265        Returns:
266            (action, distribution), or value
267
268        """
269        obs_pad = ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]
270        batch = [sample["data"] for sample in x]
271        max_len = 0
272        for sample in batch:
273            if len(sample) > max_len:
274                max_len = len(sample)
275
276        for observations in batch:
277            while len(observations) < max_len:
278                observations.append(obs_pad)
279
280        batch_embeddings = []
281        for observations in batch:
282            embeddings = []
283            for obs in observations:
284                obs_emb = self.make_embedding(obs)
285                embeddings.append(obs_emb)
286            embeddings = torch.stack(embeddings)
287            batch_embeddings.append(embeddings)
288
289        batch_embeddings = torch.stack(batch_embeddings)
290
291        lstm_out, _ = self.lstm(batch_embeddings)
292        lstm_out = lstm_out[:, -1, :]
293
294        if self.is_actor:
295            logits = self.advantage_layer(lstm_out)
296            dist = Categorical(logits=logits)
297            action = dist.sample()
298
299            return action, dist
300
301        if self.is_critic:
302            value = self.value_layer(lstm_out)
303
304            return value

307class History:
308    def __init__(
309        self,
310        block_size: int = 6,
311        action2str: dict = {0: "north", 1: "east", 2: "south", 3: "west", 4: "stay"},
312    ) -> None:
313        """Initialize the history.
314
315        Args:
316            block_size: Block size.
317            action2str: Action number to action string.
318
319        """
320        self.block_size = block_size
321        self.blocks = [[]] * self.block_size
322
323        self.action2str = action2str
324        self.actions_str = list(self.action2str.values())
325        self.actions_int = list(self.action2str.keys())
326
327    def to_list(self) -> list:
328        """Convert the history to a list."""
329        return [element for block in self.blocks for element in block]
330
331    def add_block(self, block: list) -> None:
332        """Add a block to the history."""
333        self.blocks = self.blocks[1:] + [block]
334
335    def __repr__(self) -> str:
336        return str(self.blocks)
337
338    def answer_question(self, question: list) -> str:
339        """Answer a question, by going through the history in backwards."""
340        assert len(question) == 4 and question[2] == "?"
341        for obs in self.to_list()[::-1]:
342            if obs[0] == question[0] and obs[1] == question[1]:
343                return obs[2]
344
345    def answer_questions(self, questions: list[list[str]]) -> list[str]:
346        """Answer a list of questions."""
347        return [self.answer_question(question) for question in questions]
348
349    def find_agent_current_location(self) -> str | None:
350        """Find the current location of the agent, by going through the history in
351        backwards."""
352        for obs in self.to_list()[::-1]:
353            if obs[0] == "agent" and obs[1] == "atlocation":
354                return obs[2]
355
356        return None
357
358    def explore(self, explore_policy: str) -> str:
359        """Explore the environment. The agent can either explore randomly or avoid
360        walls. This is not reinforcement learning, but a simple heuristic.
361
362        Args:
363            explore_policy: Explore policy.
364
365        Returns:
366            action_explore: Action to explore.
367
368        """
369        if explore_policy not in ["random", "avoid_walls"]:
370            raise ValueError(f"Unknown explore policy: {explore_policy}")
371
372        if explore_policy == "random":
373            action_explore = random.choice(self.actions_str)
374        elif explore_policy == "avoid_walls":
375            current_agent_location = self.find_agent_current_location()
376            if current_agent_location is None:
377                action_explore = random.choice(self.actions_str)
378            else:
379                relevant_observations = []
380                relevant_observations += [
381                    obs
382                    for obs in self.to_list()[::-1]
383                    if obs[0] == current_agent_location
384                    and obs[1] in ["north", "east", "south", "west"]
385                ]
386                # we know the agent's current location but there is no memory about map
387                if len(relevant_observations) == 0:
388                    action_explore = random.choice(self.actions_str)
389                else:
390                    # we know the agent's current location and there is at least one
391                    # memory about the map and we want to avoid the walls
392
393                    to_take = []
394                    to_avoid = []
395
396                    for obs in relevant_observations:
397                        if isinstance(obs, dict):
398                            continue
399                        if obs[2].split("_")[0] == "room":
400                            to_take.append(obs[1])
401                        elif obs[2] == "wall":
402                            if obs[1] not in to_avoid:
403                                to_avoid.append(obs[1])
404
405                    if len(to_take) > 0:
406                        action_explore = random.choice(to_take)
407                    else:
408                        options = deepcopy(self.actions_str)
409                        for e in to_avoid:
410                            options.remove(e)
411
412                        action_explore = random.choice(options)
413
414        return action_explore

566    def _run_explore_baselines(self) -> None:
567        """Run the explore baselines."""
568        env = RoomEnv2(**self.env_config)
569        observations, info = env.reset()
570        env.render("image", save_fig_dir=self.default_root_dir)
571
572        del env
573
574        env_config = deepcopy(self.env_config)
575
576        results = {}
577        for explore_policy in ["random", "avoid_walls"]:
578            results[str((explore_policy))] = []
579            scores = []
580            for test_seed in [0, 1, 2, 3, 4]:
581                score = 0
582                env_config["seed"] = test_seed
583                env = gym.make(self.env_str, **env_config)
584                observations, info = env.reset()
585                self.history = History(self.history_block_size, self.action2str)
586                self.history.add_block(observations["room"])
587
588                while True:
589                    actions_qa = self.history.answer_questions(
590                        observations["questions"]
591                    )
592                    action_explore = self.history.explore(explore_policy)
593
594                    action_pair = (actions_qa, action_explore)
595                    (
596                        observations,
597                        reward,
598                        done,
599                        truncated,
600                        info,
601                    ) = env.step(action_pair)
602                    score += reward
603                    done = done or truncated
604
605                    self.history.add_block(observations["room"])
606
607                    if done:
608                        break
609
610                scores.append(score)
611
612            results[str((explore_policy))] = {
613                "mean": np.mean(scores).item(),
614                "std": np.std(scores).item(),

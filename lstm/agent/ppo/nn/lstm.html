209    def create_embeddings(self) -> None:
210        """Create learnable embeddings."""
211        if isinstance(self.entities, dict):
212            self.word2idx = (
213                ["<PAD>"]
214                + [name for names in self.entities.values() for name in names]
215                + self.relations
216            )
217        elif isinstance(self.entities, list):
218            self.word2idx = ["<PAD>"] + self.entities + self.relations
219        else:
220            raise ValueError(
221                "entities should be either a list or a dictionary, but "
222                f"{type(self.entities)} was given!"
223            )
224        self.word2idx = {word: idx for idx, word in enumerate(self.word2idx)}
225
226        self.embeddings = nn.Embedding(
227            len(self.word2idx),
228            self.embedding_dim,
229            device=self.device,
230            padding_idx=0,
231        )
232
233        if self.make_categorical_embeddings:
234            # Assuming self.entities is a dictionary where keys are categories and
235            # values are lists of entity names
236
237            # Create a dictionary to keep track of starting index for each category
238            category_start_indices = {}
239            current_index = 1  # Start from 1 to skip the <PAD> token
240            for category, names in self.entities.items():
241                category_start_indices[category] = current_index
242                current_index += len(names)
243
244            # Re-initialize embeddings by category
245            for category, start_idx in category_start_indices.items():
246                end_idx = start_idx + len(self.entities[category])
247                init_vector = torch.randn(self.embedding_dim, device=self.device)
248                self.embeddings.weight.data[start_idx:end_idx] = init_vector.repeat(
249                    end_idx - start_idx, 1
250                )
251            # Note: Relations are not re-initialized by category, assuming they are
252            # separate from entities
253
254        if self.fuse_information == "concat":
255            self.input_size_s = self.embedding_dim * 3
256            self.input_size_e = self.embedding_dim * 3
257            self.input_size_o = self.embedding_dim * 3
258
259        elif self.fuse_information == "sum":
260            self.input_size_s = self.embedding_dim
261            self.input_size_e = self.embedding_dim
262            self.input_size_o = self.embedding_dim
263        else:
264            raise ValueError(
265                f"fuse_information should be one of 'concat' or 'sum', but "
266                f"{self.fuse_information} was given!"
267            )

269    def make_embedding(self, mem: list[str], *args) -> torch.Tensor:
270        """Create one embedding vector with summation and concatenation.
271
272        Embeddings for v1
273
274        Args:
275            mem: memory as a quadruple: [head, relation, tail, num]
276
277        Returns:
278            one embedding vector made from one memory element.
279
280        """
281        if mem == ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]:
282            if self.fuse_information == "sum":
283                return self.embeddings(
284                    torch.tensor(self.word2idx["<PAD>"], device=self.device)
285                )
286            else:
287                final_embedding = torch.concat(
288                    [
289                        self.embeddings(
290                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
291                        ),
292                        self.embeddings(
293                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
294                        ),
295                        self.embeddings(
296                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
297                        ),
298                    ]
299                )
300                return final_embedding
301
302        head_embedding = self.embeddings(
303            torch.tensor(self.word2idx[mem[0]], device=self.device)
304        )
305        relation_embedding = self.embeddings(
306            torch.tensor(self.word2idx[mem[1]], device=self.device)
307        )
308        tail_embedding = self.embeddings(
309            torch.tensor(self.word2idx[mem[2]], device=self.device)
310        )
311        if self.fuse_information == "concat":
312            final_embedding = torch.concat(
313                [head_embedding, relation_embedding, tail_embedding]
314            )
315        elif self.fuse_information == "sum":
316            final_embedding = head_embedding + relation_embedding + tail_embedding
317        else:
318            raise ValueError(
319                f"fuse_information should be one of 'concat' or 'sum', but "
320                f"{self.fuse_information} was given!"
321            )
322
323        if self.include_positional_encoding:
324            final_embedding += self.positional_encoding[mem[3]]
325
326        return final_embedding

328    def create_batch(self, x: list[list[list]], memory_type: str) -> torch.Tensor:
329        """Create one batch from data.
330
331        Args:
332            x: a batch of episodic, semantic, or short memories.
333            memory_type: "episodic", "semantic", or "short"
334
335        Returns:
336            batch of embeddings.
337
338        """
339        mem_pad = ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]
340
341        for mems in x:
342            for _ in range(self.capacity[memory_type] - len(mems)):
343                # this is a dummy entry for padding.
344                mems.append(mem_pad)
345        batch_embeddings = []
346        for mems in x:
347            embeddings = []
348            for mem in mems:
349                mem_emb = self.make_embedding(mem, memory_type)
350                embeddings.append(mem_emb)
351            embeddings = torch.stack(embeddings)
352            batch_embeddings.append(embeddings)
353
354        batch_embeddings = torch.stack(batch_embeddings)
355
356        return batch_embeddings

358    def forward(self, x_: np.ndarray) -> torch.Tensor:
359        """Forward-pass.
360
361        Note that before we make a forward pass, argument x_ will be deepcopied. This
362        is because we will modify x_ in the forward pass, and we don't want to modify
363        the original x_. This slows down the process, but it's necessary.
364
365        Args:
366            x: a batch of memories. Each element of the batch is a np.ndarray of dict
367            memories. x being a np.ndarray speeds up the process.
368
369        Returns:
370            Q-values, (action, distribution), or value.
371
372        """
373        x = deepcopy(x_)
374        assert isinstance(x, np.ndarray)
375        to_concat = []
376        if "episodic" in self.memory_of_interest:
377            batch_e = [sample["episodic"] for sample in x]  # sample is a dict
378            batch_e = self.create_batch(batch_e, memory_type="episodic")
379            lstm_out_e, _ = self.lstm_e(batch_e)
380
381            if self.fuse_information == "concat":
382                fc_out_e = self.relu(
383                    self.fc_e1(self.relu(self.fc_e0(lstm_out_e[:, -1, :])))
384                )
385                to_concat.append(fc_out_e)
386            else:
387                to_concat.append(lstm_out_e[:, -1, :])
388
389        if "episodic_agent" in self.memory_of_interest:
390            batch_e_agent = [sample["episodic_agent"] for sample in x]
391            batch_e_agent = self.create_batch(
392                batch_e_agent, memory_type="episodic_agent"
393            )
394            lstm_out_e_agent, _ = self.lstm_e_agent(batch_e_agent)
395
396            if self.fuse_information == "concat":
397                fc_out_e_agent = self.relu(
398                    self.fc_e1_agent(
399                        self.relu(self.fc_e0_agent(lstm_out_e_agent[:, -1, :]))
400                    )
401                )
402                to_concat.append(fc_out_e_agent)
403            else:
404                to_concat.append(lstm_out_e_agent[:, -1, :])
405
406        if "semantic" in self.memory_of_interest:
407            batch_s = [sample["semantic"] for sample in x]
408            batch_s = self.create_batch(batch_s, memory_type="semantic")
409            lstm_out_s, _ = self.lstm_s(batch_s)
410
411            if self.fuse_information == "concat":
412                fc_out_s = self.relu(
413                    self.fc_s1(self.relu(self.fc_s0(lstm_out_s[:, -1, :])))
414                )
415                to_concat.append(fc_out_s)
416            else:
417                to_concat.append(lstm_out_s[:, -1, :])
418
419        if "semantic_map" in self.memory_of_interest:
420            batch_s_map = [sample["semantic_map"] for sample in x]
421            batch_s_map = self.create_batch(batch_s_map, memory_type="semantic_map")
422            lstm_out_s_map, _ = self.lstm_s_map(batch_s_map)
423
424            if self.fuse_information == "concat":
425                fc_out_s_map = self.relu(
426                    self.fc_s1_map(self.relu(self.fc_s0_map(lstm_out_s_map[:, -1, :])))
427                )
428                to_concat.append(fc_out_s_map)
429            else:
430                to_concat.append(lstm_out_s_map[:, -1, :])
431
432        if "short" in self.memory_of_interest:
433            batch_o = [sample["short"] for sample in x]
434            batch_o = self.create_batch(batch_o, memory_type="short")
435            lstm_out_o, _ = self.lstm_o(batch_o)
436
437            if self.fuse_information == "concat":
438                fc_out_o = self.relu(
439                    self.fc_o1(self.relu(self.fc_o0(lstm_out_o[:, -1, :])))
440                )
441                to_concat.append(fc_out_o)
442            else:
443                to_concat.append(lstm_out_o[:, -1, :])
444
445        if self.fuse_information == "concat":
446            fc_out_all = torch.concat(to_concat, dim=-1)
447        elif self.fuse_information == "sum":
448            fc_out_all = torch.sum(torch.stack(to_concat), dim=0)
449        else:
450            raise ValueError(
451                f"fuse_information should be one of 'concat' or 'sum', but "
452                f"{self.fuse_information} was given!"
453            )
454
455        if self.is_actor:
456            logits = self.action_layer(fc_out_all)
457            dist = Categorical(logits=logits)
458            action = dist.sample()
459
460            return action, dist
461
462        if self.is_critic:
463            value = self.value_layer(fc_out_all)
464
465            return value

210    def create_embeddings(self) -> None:
211        """Create learnable embeddings."""
212        if isinstance(self.entities, dict):
213            self.word2idx = (
214                ["<PAD>"]
215                + [name for names in self.entities.values() for name in names]
216                + self.relations
217            )
218        elif isinstance(self.entities, list):
219            self.word2idx = ["<PAD>"] + self.entities + self.relations
220        else:
221            raise ValueError(
222                "entities should be either a list or a dictionary, but "
223                f"{type(self.entities)} was given!"
224            )
225        self.word2idx = {word: idx for idx, word in enumerate(self.word2idx)}
226
227        self.embeddings = nn.Embedding(
228            len(self.word2idx),
229            self.embedding_dim,
230            device=self.device,
231            padding_idx=0,
232        )
233
234        if self.make_categorical_embeddings:
235            # Assuming self.entities is a dictionary where keys are categories and
236            # values are lists of entity names
237
238            # Create a dictionary to keep track of starting index for each category
239            category_start_indices = {}
240            current_index = 1  # Start from 1 to skip the <PAD> token
241            for category, names in self.entities.items():
242                category_start_indices[category] = current_index
243                current_index += len(names)
244
245            # Re-initialize embeddings by category
246            for category, start_idx in category_start_indices.items():
247                end_idx = start_idx + len(self.entities[category])
248                init_vector = torch.randn(self.embedding_dim, device=self.device)
249                self.embeddings.weight.data[start_idx:end_idx] = init_vector.repeat(
250                    end_idx - start_idx, 1
251                )
252            # Note: Relations are not re-initialized by category, assuming they are
253            # separate from entities
254
255        if self.fuse_information == "concat":
256            self.input_size_s = self.embedding_dim * 3
257            self.input_size_e = self.embedding_dim * 3
258            self.input_size_o = self.embedding_dim * 3
259
260        elif self.fuse_information == "sum":
261            self.input_size_s = self.embedding_dim
262            self.input_size_e = self.embedding_dim
263            self.input_size_o = self.embedding_dim
264        else:
265            raise ValueError(
266                f"fuse_information should be one of 'concat' or 'sum', but "
267                f"{self.fuse_information} was given!"

270    def make_embedding(self, mem: list[str], *args) -> torch.Tensor:
271        """Create one embedding vector with summation and concatenation.
272
273        Embeddings for v1
274
275        Args:
276            mem: memory as a quadruple: [head, relation, tail, num]
277
278        Returns:
279            one embedding vector made from one memory element.
280
281        """
282        if mem == ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]:
283            if self.fuse_information == "sum":
284                return self.embeddings(
285                    torch.tensor(self.word2idx["<PAD>"], device=self.device)
286                )
287            else:
288                final_embedding = torch.concat(
289                    [
290                        self.embeddings(
291                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
292                        ),
293                        self.embeddings(
294                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
295                        ),
296                        self.embeddings(
297                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
298                        ),
299                    ]
300                )
301                return final_embedding
302
303        head_embedding = self.embeddings(
304            torch.tensor(self.word2idx[mem[0]], device=self.device)
305        )
306        relation_embedding = self.embeddings(
307            torch.tensor(self.word2idx[mem[1]], device=self.device)
308        )
309        tail_embedding = self.embeddings(
310            torch.tensor(self.word2idx[mem[2]], device=self.device)
311        )
312        if self.fuse_information == "concat":
313            final_embedding = torch.concat(
314                [head_embedding, relation_embedding, tail_embedding]
315            )
316        elif self.fuse_information == "sum":
317            final_embedding = head_embedding + relation_embedding + tail_embedding
318        else:
319            raise ValueError(
320                f"fuse_information should be one of 'concat' or 'sum', but "
321                f"{self.fuse_information} was given!"
322            )
323
324        if self.include_positional_encoding:
325            final_embedding += self.positional_encoding[mem[3]]
326
327        return final_embedding

329    def create_batch(self, x: list[list[list]], memory_type: str) -> torch.Tensor:
330        """Create one batch from data.
331
332        Args:
333            x: a batch of episodic, semantic, or short memories.
334            memory_type: "episodic", "semantic", or "short"
335
336        Returns:
337            batch of embeddings.
338
339        """
340        mem_pad = ["<PAD>", "<PAD>", "<PAD>", "<PAD>"]
341
342        for mems in x:
343            for _ in range(self.capacity[memory_type] - len(mems)):
344                # this is a dummy entry for padding.
345                mems.append(mem_pad)
346        batch_embeddings = []
347        for mems in x:
348            embeddings = []
349            for mem in mems:
350                mem_emb = self.make_embedding(mem, memory_type)
351                embeddings.append(mem_emb)
352            embeddings = torch.stack(embeddings)
353            batch_embeddings.append(embeddings)
354
355        batch_embeddings = torch.stack(batch_embeddings)
356
357        return batch_embeddings

359    def forward(self, x_: np.ndarray) -> torch.Tensor:
360        """Forward-pass.
361
362        Note that before we make a forward pass, argument x_ will be deepcopied. This
363        is because we will modify x_ in the forward pass, and we don't want to modify
364        the original x_. This slows down the process, but it's necessary.
365
366        Args:
367            x: a batch of memories. Each element of the batch is a np.ndarray of dict
368            memories. x being a np.ndarray speeds up the process.
369
370        Returns:
371            Q-values, (action, distribution), or value.
372
373        """
374        x = deepcopy(x_)
375        assert isinstance(x, np.ndarray)
376        to_concat = []
377        if "episodic" in self.memory_of_interest:
378            batch_e = [sample["episodic"] for sample in x]  # sample is a dict
379            batch_e = self.create_batch(batch_e, memory_type="episodic")
380            lstm_out_e, _ = self.lstm_e(batch_e)
381
382            if self.fuse_information == "concat":
383                fc_out_e = self.relu(
384                    self.fc_e1(self.relu(self.fc_e0(lstm_out_e[:, -1, :])))
385                )
386                to_concat.append(fc_out_e)
387            else:
388                to_concat.append(lstm_out_e[:, -1, :])
389
390        if "episodic_agent" in self.memory_of_interest:
391            batch_e_agent = [sample["episodic_agent"] for sample in x]
392            batch_e_agent = self.create_batch(
393                batch_e_agent, memory_type="episodic_agent"
394            )
395            lstm_out_e_agent, _ = self.lstm_e_agent(batch_e_agent)
396
397            if self.fuse_information == "concat":
398                fc_out_e_agent = self.relu(
399                    self.fc_e1_agent(
400                        self.relu(self.fc_e0_agent(lstm_out_e_agent[:, -1, :]))
401                    )
402                )
403                to_concat.append(fc_out_e_agent)
404            else:
405                to_concat.append(lstm_out_e_agent[:, -1, :])
406
407        if "semantic" in self.memory_of_interest:
408            batch_s = [sample["semantic"] for sample in x]
409            batch_s = self.create_batch(batch_s, memory_type="semantic")
410            lstm_out_s, _ = self.lstm_s(batch_s)
411
412            if self.fuse_information == "concat":
413                fc_out_s = self.relu(
414                    self.fc_s1(self.relu(self.fc_s0(lstm_out_s[:, -1, :])))
415                )
416                to_concat.append(fc_out_s)
417            else:
418                to_concat.append(lstm_out_s[:, -1, :])
419
420        if "semantic_map" in self.memory_of_interest:
421            batch_s_map = [sample["semantic_map"] for sample in x]
422            batch_s_map = self.create_batch(batch_s_map, memory_type="semantic_map")
423            lstm_out_s_map, _ = self.lstm_s_map(batch_s_map)
424
425            if self.fuse_information == "concat":
426                fc_out_s_map = self.relu(
427                    self.fc_s1_map(self.relu(self.fc_s0_map(lstm_out_s_map[:, -1, :])))
428                )
429                to_concat.append(fc_out_s_map)
430            else:
431                to_concat.append(lstm_out_s_map[:, -1, :])
432
433        if "short" in self.memory_of_interest:
434            batch_o = [sample["short"] for sample in x]
435            batch_o = self.create_batch(batch_o, memory_type="short")
436            lstm_out_o, _ = self.lstm_o(batch_o)
437
438            if self.fuse_information == "concat":
439                fc_out_o = self.relu(
440                    self.fc_o1(self.relu(self.fc_o0(lstm_out_o[:, -1, :])))
441                )
442                to_concat.append(fc_out_o)
443            else:
444                to_concat.append(lstm_out_o[:, -1, :])
445
446        if self.fuse_information == "concat":
447            fc_out_all = torch.concat(to_concat, dim=-1)
448        elif self.fuse_information == "sum":
449            fc_out_all = torch.sum(torch.stack(to_concat), dim=0)
450        else:
451            raise ValueError(
452                f"fuse_information should be one of 'concat' or 'sum', but "
453                f"{self.fuse_information} was given!"
454            )
455
456        if self.is_actor:
457            logits = self.action_layer(fc_out_all)
458            dist = Categorical(logits=logits)
459            action = dist.sample()
460
461            return action, dist
462
463        if self.is_critic:
464            value = self.value_layer(fc_out_all)
465
466            return value

289                    [
290                        self.embeddings(
291                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
292                        ),

294                            torch.tensor(self.word2idx["<PAD>"], device=self.device)
295                        ),
296                        self.embeddings(

298                        ),

304            torch.tensor(self.word2idx[mem[0]], device=self.device)
305        )
306        relation_embedding = self.embeddings(
307            torch.tensor(self.word2idx[mem[1]], device=self.device)
308        )
309        tail_embedding = self.embeddings(
310            torch.tensor(self.word2idx[mem[2]], device=self.device)
311        )
312        if self.fuse_information == "concat":
313            final_embedding = torch.concat(
314                [head_embedding, relation_embedding, tail_embedding]
315            )
316        elif self.fuse_information == "sum":
317            final_embedding = head_embedding + relation_embedding + tail_embedding
318        else:
319            raise ValueError(
320                f"fuse_information should be one of 'concat' or 'sum', but "
321                f"{self.fuse_information} was given!"
322            )

325            final_embedding += self.positional_encoding[mem[3]]
326

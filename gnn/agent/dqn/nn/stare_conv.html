 81    def forward(
 82        self,
 83        entity_embeddings: torch.Tensor,
 84        relation_embeddings: torch.Tensor,
 85        edge_idx: torch.Tensor,
 86        edge_type: torch.Tensor,
 87        quals: torch.Tensor,
 88    ) -> Tuple[torch.Tensor, torch.Tensor]:
 89        """Perform forward pass of the StarE convolution layer.
 90
 91        `entity_embeddings` is a tensor of shape [num_nodes, emb_dim], which is a stack
 92        of entity embeddings. Since x is a sampled sub-graph of the entire hidden graph,
 93        `num_nodes` is always less than the total number of global entities.
 94        `edge_idx` is a tensor of shape [2, num_edges] and `edge_type` is a tensor
 95        of shape [num_edges]. `relation_embeddings` is a tensor of shape
 96        [num_rels, emb_dim].
 97
 98
 99        Args:
100            entity_embeddings: Node (the entities in a given graph) feature matrix.
101            relation_embeddings: Relation embeddings.
102            edge_idx: Graph edge indices.
103            edge_type: Edge type indices.
104            quals: Qualifier indices.
105
106        Returns:
107            Output node features and relation embeddings.
108        """
109        if self.device is None:
110            self.device = edge_idx.device
111
112        rel_embed = torch.cat([relation_embeddings, self.loop_rel], dim=0)
113        num_edges = edge_idx.size(1) // 2
114        num_ent = entity_embeddings.size(0)
115
116        self.in_index, self.out_index = (
117            edge_idx[:, :num_edges],
118            edge_idx[:, num_edges:],
119        )
120        self.in_type, self.out_type = edge_type[:num_edges], edge_type[num_edges:]
121
122        num_quals = quals.size(1) // 2
123        self.in_index_qual_ent, self.out_index_qual_ent = (
124            quals[1, :num_quals],
125            quals[1, num_quals:],
126        )
127        self.in_index_qual_rel, self.out_index_qual_rel = (
128            quals[0, :num_quals],
129            quals[0, num_quals:],
130        )
131        self.quals_index_in, self.quals_index_out = (
132            quals[2, :num_quals],
133            quals[2, num_quals:],
134        )
135
136        # Self edges between all the nodes
137        self.loop_index = torch.stack(
138            [torch.arange(num_ent), torch.arange(num_ent)]
139        ).to(self.device)
140        self.loop_type = torch.full(
141            (num_ent,), rel_embed.size(0) - 1, dtype=torch.long
142        ).to(self.device)
143
144        self.in_norm = self.compute_norm(self.in_index, num_ent)
145        self.out_norm = self.compute_norm(self.out_index, num_ent)
146
147        in_res = self.propagate(
148            self.in_index,
149            x=entity_embeddings,
150            edge_type=self.in_type,
151            rel_embed=rel_embed,
152            edge_norm=self.in_norm,
153            mode="in",
154            ent_embed=entity_embeddings,
155            qualifier_ent=self.in_index_qual_ent,
156            qualifier_rel=self.in_index_qual_rel,
157            qual_index=self.quals_index_in,
158            source_index=self.in_index[0],
159        )
160
161        loop_res = self.propagate(
162            self.loop_index,
163            x=entity_embeddings,
164            edge_type=self.loop_type,
165            rel_embed=rel_embed,
166            edge_norm=None,
167            mode="loop",
168            ent_embed=None,
169            qualifier_ent=None,
170            qualifier_rel=None,
171            qual_index=None,
172            source_index=None,
173        )
174        out_res = self.propagate(
175            self.out_index,
176            x=entity_embeddings,
177            edge_type=self.out_type,
178            rel_embed=rel_embed,
179            edge_norm=self.out_norm,
180            mode="out",
181            ent_embed=entity_embeddings,
182            qualifier_ent=self.out_index_qual_ent,
183            qualifier_rel=self.out_index_qual_rel,
184            qual_index=self.quals_index_out,
185            source_index=self.out_index[0],
186        )
187
188        out = (
189            self.drop(in_res) * (1 / 3)
190            + self.drop(out_res) * (1 / 3)
191            + loop_res * (1 / 3)
192        )
193
194        out = self.bn(out)
195
196        # Ignoring the self loop inserted, return.
197        return torch.tanh(out), torch.matmul(rel_embed, self.w_rel)[:-1]

229    def qualifier_aggregate(
230        self,
231        qualifier_emb: torch.Tensor,
232        rel_part_emb: torch.Tensor,
233        alpha: float,
234        qual_index: Optional[torch.Tensor] = None,
235    ) -> torch.Tensor:
236        """Aggregate qualifier embeddings.
237
238        Args:
239            qualifier_emb: Qualifier embeddings.
240            rel_part_emb: Relation part embeddings.
241            alpha: Weight factor for aggregation.
242            qual_index: Qualifier indices.
243
244        Returns:
245            Aggregated embeddings.
246        """
247        qualifier_emb = torch.einsum(
248            "ij,jk -> ik",
249            self.coalesce_quals(qualifier_emb, qual_index, rel_part_emb.shape[0]),
250            self.w_q,
251        )
252
253        return alpha * rel_part_emb + (1 - alpha) * qualifier_emb

255    def update_rel_emb_with_qualifier(
256        self,
257        ent_embed: torch.Tensor,
258        rel_embed: torch.Tensor,
259        qualifier_ent: torch.Tensor,
260        qualifier_rel: torch.Tensor,
261        edge_type: torch.Tensor,
262        qual_index: Optional[torch.Tensor] = None,
263    ) -> torch.Tensor:
264        """Update relation embeddings with qualifier embeddings.
265
266        Args:
267            ent_embed: Entity embeddings.
268            rel_embed: Relation embeddings.
269            qualifier_ent: Qualifier entity embeddings.
270            qualifier_rel: Qualifier relation embeddings.
271            edge_type: Edge type indices.
272            qual_index: Qualifier indices.
273
274        Returns:
275            Updated relation embeddings.
276        """
277
278        # Step 1: embedding
279        qualifier_emb_rel = rel_embed[qualifier_rel]
280        qualifier_emb_ent = ent_embed[qualifier_ent]
281
282        rel_part_emb = rel_embed[edge_type]
283
284        # Step 2: pass it through qual_transform
285        qualifier_emb = self.qual_transform(
286            qualifier_ent=qualifier_emb_ent, qualifier_rel=qualifier_emb_rel
287        )
288
289        # Pass it through an aggregate layer
290        return self.qualifier_aggregate(
291            qualifier_emb,
292            rel_part_emb,
293            alpha=self.triple_qual_weight,
294            qual_index=qual_index,
295        )

297    def message(
298        self,
299        x_j: torch.Tensor,
300        x_i: torch.Tensor,
301        edge_type: torch.Tensor,
302        rel_embed: torch.Tensor,
303        edge_norm: Optional[torch.Tensor],
304        mode: str,
305        ent_embed: Optional[torch.Tensor] = None,
306        qualifier_ent: Optional[torch.Tensor] = None,
307        qualifier_rel: Optional[torch.Tensor] = None,
308        qual_index: Optional[torch.Tensor] = None,
309        source_index: Optional[torch.Tensor] = None,
310    ) -> torch.Tensor:
311        """Construct messages for message passing.
312
313        Args:
314            x_j: Source node features.
315            x_i: Target node features.
316            edge_type: Edge type indices.
317            rel_embed: Relation embeddings.
318            edge_norm: Edge normalization coefficients.
319            mode (str): Mode of message passing (in, out, loop).
320            ent_embed: Entity embeddings.
321            qualifier_ent: Qualifier entity embeddings.
322            qualifier_rel: Qualifier relation embeddings.
323            qual_index: Qualifier indices.
324            source_index: Source node indices.
325
326        Returns:
327            Messages for aggregation.
328        """
329        weight = getattr(self, "w_{}".format(mode))
330
331        if mode != "loop":
332            rel_emb = self.update_rel_emb_with_qualifier(
333                ent_embed,
334                rel_embed,
335                qualifier_ent,
336                qualifier_rel,
337                edge_type,
338                qual_index,
339            )
340        else:
341            rel_emb = torch.index_select(rel_embed, 0, edge_type)
342
343        xj_rel = self.rel_transform(x_j, rel_emb)
344        out = torch.einsum("ij,jk->ik", xj_rel, weight)
345
346        return out if edge_norm is None else out * edge_norm.view(-1, 1)
347

360    @staticmethod
361    def compute_norm(edge_idx: torch.Tensor, num_ent: int) -> torch.Tensor:
362        """Compute normalization coefficients for edges.
363
364        Args:
365            edge_idx: Edge indices.
366            num_ent: Number of entities.
367
368        Returns:
369            Normalization coefficients.
370        """
371        row, col = edge_idx
372        edge_weight = torch.ones_like(row).float()
373        deg = scatter_add(edge_weight, row, dim=0, dim_size=num_ent)
374        deg_inv = deg.pow(-0.5)
375        deg_inv[deg_inv == float("inf")] = 0
376        norm = deg_inv[row] * edge_weight * deg_inv[col]
377        return norm

379    def coalesce_quals(
380        self,
381        qual_embeddings: torch.Tensor,
382        qual_index: torch.Tensor,
383        num_edges: int,
384        fill: int = 0,
385    ) -> torch.Tensor:
386        """Coalesce qualifier embeddings.
387
388        Args:
389            qual_embeddings: Qualifier embeddings.
390            qual_index: Qualifier indices.
391            num_edges: Number of edges.
392            fill: Fill value for empty embeddings. Default is 0.
393
394        Returns:
395            Coalesced qualifier embeddings.
396        """
397        output = scatter_add(qual_embeddings, qual_index, dim=0, dim_size=num_edges)
398
399        if fill != 0:
400            mask = output.sum(dim=-1) == 0
401            output[mask] = fill
402
403        return output

 86        edge_idx: torch.Tensor,
 87        edge_type: torch.Tensor,
 88        quals: torch.Tensor,
 89    ) -> Tuple[torch.Tensor, torch.Tensor]:
 90        """Perform forward pass of the StarE convolution layer.
 91
 92        `entity_embeddings` is a tensor of shape [num_nodes, emb_dim], which is a stack
 93        of entity embeddings. Since x is a sampled sub-graph of the entire hidden graph,
 94        `num_nodes` is always less than the total number of global entities.
 95        `edge_idx` is a tensor of shape [2, num_edges] and `edge_type` is a tensor
 96        of shape [num_edges]. `relation_embeddings` is a tensor of shape
 97        [num_rels, emb_dim].
 98
 99
100        Args:
101            entity_embeddings: Node (the entities in a given graph) feature matrix.
102            relation_embeddings: Relation embeddings.
103            edge_idx: Graph edge indices.
104            edge_type: Edge type indices.
105            quals: Qualifier indices.
106
107        Returns:
108            Output node features and relation embeddings.
109        """
110        if self.device is None:
111            self.device = edge_idx.device
112
113        rel_embed = torch.cat([relation_embeddings, self.loop_rel], dim=0)
114        num_edges = edge_idx.size(1) // 2
115        num_ent = entity_embeddings.size(0)
116
117        self.in_index, self.out_index = (
118            edge_idx[:, :num_edges],
119            edge_idx[:, num_edges:],
120        )
121        self.in_type, self.out_type = edge_type[:num_edges], edge_type[num_edges:]
122
123        num_quals = quals.size(1) // 2
124        self.in_index_qual_ent, self.out_index_qual_ent = (
125            quals[1, :num_quals],
126            quals[1, num_quals:],
127        )
128        self.in_index_qual_rel, self.out_index_qual_rel = (
129            quals[0, :num_quals],
130            quals[0, num_quals:],
131        )
132        self.quals_index_in, self.quals_index_out = (
133            quals[2, :num_quals],
134            quals[2, num_quals:],
135        )
136
137        # Self edges between all the nodes
138        self.loop_index = torch.stack(
139            [torch.arange(num_ent), torch.arange(num_ent)]
140        ).to(self.device)
141        self.loop_type = torch.full(
142            (num_ent,), rel_embed.size(0) - 1, dtype=torch.long
143        ).to(self.device)
144
145        self.in_norm = self.compute_norm(self.in_index, num_ent)
146        self.out_norm = self.compute_norm(self.out_index, num_ent)
147
148        in_res = self.propagate(
149            self.in_index,
150            x=entity_embeddings,
151            edge_type=self.in_type,
152            rel_embed=rel_embed,
153            edge_norm=self.in_norm,
154            mode="in",
155            ent_embed=entity_embeddings,
156            qualifier_ent=self.in_index_qual_ent,
157            qualifier_rel=self.in_index_qual_rel,
158            qual_index=self.quals_index_in,
159            source_index=self.in_index[0],
160        )
161
162        loop_res = self.propagate(
163            self.loop_index,
164            x=entity_embeddings,
165            edge_type=self.loop_type,
166            rel_embed=rel_embed,
167            edge_norm=None,
168            mode="loop",
169            ent_embed=None,
170            qualifier_ent=None,
171            qualifier_rel=None,
172            qual_index=None,
173            source_index=None,
174        )
175        out_res = self.propagate(
176            self.out_index,
177            x=entity_embeddings,
178            edge_type=self.out_type,
179            rel_embed=rel_embed,
180            edge_norm=self.out_norm,
181            mode="out",
182            ent_embed=entity_embeddings,
183            qualifier_ent=self.out_index_qual_ent,
184            qualifier_rel=self.out_index_qual_rel,
185            qual_index=self.quals_index_out,
186            source_index=self.out_index[0],
187        )
188
189        out = (
190            self.drop(in_res) * (1 / 3)
191            + self.drop(out_res) * (1 / 3)
192            + loop_res * (1 / 3)
193        )
194
195        out = self.bn(out)
196
197        # Ignoring the self loop inserted, return.
198        return torch.tanh(out), torch.matmul(rel_embed, self.w_rel)[:-1]
